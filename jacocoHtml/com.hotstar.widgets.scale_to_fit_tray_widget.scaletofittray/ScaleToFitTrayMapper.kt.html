package com.hotstar.widgets.scale_to_fit_tray_widget.scaletofittray

import androidx.compose.runtime.Composable
import androidx.compose.ui.Modifier
import com.hotstar.bff.models.widget.BffRectangleStudioWidget
import com.hotstar.bff.models.widget.BffScaleToFitTrayItemWidget
import com.hotstar.bff.models.widget.BffSquareStudioWidget
import com.hotstar.widgets.scale_to_fit_tray_widget.widgetitems.StudioTileUi

@Composable
fun ScaleToFitTrayMapper(
    modifier: Modifier = Modifier,
    widgetItem: BffScaleToFitTrayItemWidget
) {
    when (widgetItem) {
        is BffSquareStudioWidget -> {
            StudioTileUi(
                imageData = widgetItem.image,
                action = widgetItem.action,
                widgetCommons = widgetItem.widgetCommons,
                modifier = modifier,
                widgetItem = widgetItem
            )
        }
        is BffRectangleStudioWidget -> {
            StudioTileUi(
                imageData = widgetItem.image,
                action = widgetItem.action,
                widgetCommons = widgetItem.widgetCommons,
                modifier = modifier,
                widgetItem = widgetItem
            )
        }
    }
}
